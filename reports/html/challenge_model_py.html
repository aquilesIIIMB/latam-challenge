18 """ 

19 A model to predict flight delays using XGBoost classifier.

26 self

31 # Top 10 features based on feature importance analysis

33 "OPERA_Latin American Wings", 

34 "MES_7", 

35 "MES_10", 

36 "MES_12", 

37 "MES_8", 

38 "OPERA_JetSmart SPA",

46 def preprocess( 

47 self, 

48 data: pd.DataFrame, 

49 target_column: str = None 

50 ) -> Union[Tuple[pd.DataFrame, pd.DataFrame], pd.DataFrame]: 

51 """ 

52 Prepare raw data for training or prediction. 

53 

54 Args: 

55 data (pd.DataFrame): Raw data to preprocess. 

56 target_column (str, optional): If set, the target is returned. 

57 

58 Returns: 

59 Union[Tuple[pd.DataFrame, pd.DataFrame], pd.DataFrame]:  

60 Features and target (if target_column is provided), 

61 or just features. 

62 """ 

63 logger.info("Starting data preprocessing") 

64 

65 try: 

66 def get_min_diff(data): 

67 fecha_o = datetime.strptime(data['Fecha-O'], '%Y-%m-%d %H:%M:%S') 

68 fecha_i = datetime.strptime(data['Fecha-I'], '%Y-%m-%d %H:%M:%S') 

69 min_diff = ((fecha_o - fecha_i).total_seconds())/60 

70 return min_diff 

71 

72 data['min_diff'] = data.apply(get_min_diff, axis = 1) 

73 

74 threshold_in_minutes = 15 

75 data['delay'] = np.where(data['min_diff'] > threshold_in_minutes, 1, 0) 

76 

77 # Create features using one-hot encoding for categorical variables 

78 features = pd.concat([ 

79 pd.get_dummies(data['OPERA'], prefix='OPERA'), 

80 pd.get_dummies(data['TIPOVUELO'], prefix='TIPOVUELO'), 

81 pd.get_dummies(data['MES'], prefix='MES') 

82 ], axis=1) 

83 

84 logger.info(f"Created {features.shape[1]} features through one-hot encoding") 

85 

86 # Ensure all top features are present (fill with zeros if missing) 

87 for feature in self._top_features: 

88 if feature not in features.columns: 

89 features[feature] = 0 

90 logger.info(f"Added missing feature column: {feature}") 

91 

92 # Select only the top features for the model 

93 features = features[self._top_features] 

94 logger.info(f"Selected {features.shape[1]} top features for model") 

95 

96 # If target column is provided, return features and target 

97 if target_column is not None: 

98 if target_column in data.columns: 

99 # Prepare target as a DataFrame with delay column 

100 target = pd.DataFrame(data[target_column]).copy() 

101 logger.info(f"Target column '{target_column}' prepared") 

102 return features, target 

103 else: 

104 logger.error(f"Target column '{target_column}' not found in data") 

105 raise ValueError(f"Target column '{target_column}' not found in data") 

106 

107 return features 

108 

109 except Exception as e: 

110 logger.error(f"Error during preprocessing: {str(e)}") 

111 raise

125 logger.info("Starting model fitting") 

126 

127 try: 

128 # Calculate class balance for weighting 

129 n_y0 = len(target[target.iloc[:, 0] == 0]) # Number of non-delayed flights 

130 n_y1 = len(target[target.iloc[:, 0] == 1]) # Number of delayed flights 

131 

132 if n_y1 == 0: 

133 logger.warning("No delayed flights in training data") 

134 scale = 1.0 

135 else: 

136 scale = n_y0 / n_y1 # Weight for positive class 

137 logger.info(f"Class imbalance ratio: {scale:.2f} (non-delayed/delayed)") 

138 

139 # Initialize XGBoost classifier with balanced class weights 

140 self._model = xgb.XGBClassifier( 

141 random_state=1, 

142 learning_rate=0.01, 

143 scale_pos_weight=scale 

144 ) 

145 

146 # Perform train-test split for evaluation 

147 X_train, X_test, y_train, y_test = train_test_split( 

148 features, target.values.ravel(), test_size=0.2, random_state=42 

149 ) 

150 

151 logger.info(f"Training on {X_train.shape[0]} samples, evaluating on {X_test.shape[0]} samples") 

152 

153 # Train the model on the training split 

154 self._model.fit(X_train, y_train) 

155 

156 # Evaluate the model on the test split 

157 y_pred = self._model.predict(X_test) 

158 

159 # Log confusion matrix 

160 cm = confusion_matrix(y_test, y_pred) 

161 logger.info(f"Confusion Matrix:\n{cm}") 

162 

163 # Log classification report 

164 report = classification_report(y_test, y_pred, output_dict=True) 

165 logger.info("Classification Report:") 

166 logger.info(f"Class 0 - Precision: {report['0']['precision']:.2f}, Recall: {report['0']['recall']:.2f}, F1: {report['0']['f1-score']:.2f}") 

167 logger.info(f"Class 1 - Precision: {report['1']['precision']:.2f}, Recall: {report['1']['recall']:.2f}, F1: {report['1']['f1-score']:.2f}") 

168 logger.info(f"Accuracy: {report['accuracy']:.2f}") 

169 

170 # Now train the final model on all data 

171 logger.info("Training final model on complete dataset") 

172 self._model.fit(features, target.values.ravel()) 

173 

174 logger.info("Model successfully trained") 

175 

176 except Exception as e:
